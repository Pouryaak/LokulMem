       - status: StorageStatus (initialized to isReadOnly: false, lastError: null, dbVersion: 0, isOpen: false)
       - onStorageError?: (error: StorageError) => void

    3. Constructor(options?: { onStorageError?: (error: StorageError) => void }):
       - Create new LokulDatabase instance
       - Store callback if provided

    4. async initialize(): Promise<void>:
       - Try to open database with this.db.open()
       - On success: update status.isOpen = true, status.dbVersion = this.db.verno
       - On error: call handleStorageError() with appropriate error type
       - Specifically catch:
         - Dexie.QuotaExceededError -> set read-only mode
         - Dexie.DatabaseClosedError, Dexie.OpenFailedError -> corruption
         - Other Dexie errors -> migration_failed or unknown

    5. private handleStorageError(error: Error): void:
       - Determine StorageErrorType based on error type
       - Update status.lastError and status.isReadOnly (if quota exceeded)
       - Call onStorageError callback if provided
       - Log error for debugging

    6. async getVersion(): Promise<number>:
       - Return this.db.verno (throws if db not open)

    7. getStatus(): StorageStatus:
       - Return copy of current status object

    8. async close(): Promise<void>:
       - Close database connection
       - Update status.isOpen = false

    9. isReadOnly(): boolean:
       - Return status.isReadOnly

    10. getDb(): LokulDatabase:
        - Return the database instance (for repository use)

    Add JSDoc for all public methods explaining their purpose and error conditions.
